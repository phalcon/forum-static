Try this.

public function updateResponse($state, $sugester, $sugested, $location_id) {
    $Sugestion = Sugestion::findFirst("sugester = '{$sugester}' AND sugested = '{$sugested}' AND location_id = '{$location_id}'");
    $Sugestion->state = $state;
    $Sugestion->save();
}
