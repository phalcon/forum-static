Try to see what messages are being returning!

$query = "UPDATE media SET media.mid = 2 WHERE media.name LIKE test";
$status = $this->getModelsManager()->executeQuery($query);
if (!$status->success()) {
    foreach ($status->getMessages() as $message) {
        echo $message;
    }
}
