You can define two routes:

$router->add(
    "/movie/detail/{id}/{name}/{importFlag}",
    array(
        "controller" => "movies",
        "action"     => "getDetails",
    )
);

$router->add(
    "/movie/detail/{id}/{name}",
    array(
        "controller" => "movies",
        "action"     => "getDetails",
    )
);
