I find solutions edit my relations defenition

public function initialize()
    {
        $this->hasManyToMany(
            "id",
            "\ProductsAttributes",
            "product_id",
            "attribute_id",
            "\Attributes",
            'id',
            [
                'alias'=>'attributes',
                'params'=>
                    [
                        //Add array and define models name in square brackets
                        'columns'=>[
                            '[\Attributes].*', 
                             '[\ProductsAttributes].attributes_values AS vals '
                            ]
                    ]
            ]
        );
    }
