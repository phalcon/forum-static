You have to adjust your baseUri according to the host:

$di->set('url', function(){
        $url = new \Phalcon\Mvc\Url();
        $url->setBaseUri('/'); // https://myapp.dev
        $url->setBaseUri('/myapp/'); // https://localhost/myapp/
        return $url;
});
