now yes but you can overload the Resultset::toArray(); with something like this

class MyModel extends \Phalcon\Mvc\Model {
    public function toArray() {
        $result = [];
        foreach($this->_rows as $row) {
            $result[] = $row->toArray();
        }
        return $result;
    }
}
