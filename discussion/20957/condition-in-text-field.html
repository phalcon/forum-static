Maybe determine that before the field:

{% raw %}{%{% endraw %} set required = true {% raw %}%}{% endraw %}
{% raw %}{%{% endraw %} if contract.id is not empty {% raw %}%}{% endraw %}
    {% raw %}{%{% endraw %} set required = false {% raw %}%}{% endraw %}
{% raw %}{%{% endraw %} endif {% raw %}%}{% endraw %}

{% raw %}{{{% endraw %} text_field('test', 'class': 'input w-full border', 'value': contract.totalPrice|default(""), 'required': required) {% raw %}}}{% endraw %}

Or, if you don't mind mixing in a little raw PHP code:

<?php $required = !isset($contract.id); ?>

{% raw %}{{{% endraw %} text_field('test', 'class': 'input w-full border', 'value': contract.totalPrice|default(""), 'required': required) {% raw %}}}{% endraw %}
