FYI Redirect does not change the execution flow, you need: 

public function logoutAction()
    {
        $this->auth->logout();
        $this->session->destroy();
        $this->flash->success("User disconnects.");
        return $this->response->redirect('/');
        die("Why is this happening");
    }
