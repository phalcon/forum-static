This way:

$this->modelsManager->createBuilder()
    ->from(array('g' => 'Goods'))
    ->join('Enterprises','g.enterprise_id = e.id','e')
    ->where('g.category_id = :id:',array('id' => $category_id))
    ->andWhere("e.full_name like :keyword:",array('keyword' => '%'. $keyword. '%'))
    ->getQuery()
    ->execute();
