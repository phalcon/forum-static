I'm not sure if this is what you mean, but the order in which you put the fields is how they get rendered. You could also then use Volt function to create the mark-up and group the items.

Here's an example how i do it:

{% raw %}{%{% endraw %} block content {% raw %}%}{% endraw %}
<div class="box box-success">
    <div class="box-header width-border">
        <h3 class="box-title">Form</h3>
    </div>

    <div class="box-body">
        <div class="row">
            <div class="col-xs-12">
                {% raw %}{{{% endraw %} form("client/save") {% raw %}}}{% endraw %}
                    {% raw %}{%{% endraw %} for element in form {% raw %}%}{% endraw %}
                        <div class="form-group">
                            {% raw %}{{{% endraw %} element.label(["class": "col-sm-3 control-label"]) {% raw %}}}{% endraw %}
                            <div class="col-sm-9">
                                {% raw %}{{{% endraw %} element {% raw %}}}{% endraw %}
                            </div>
                        </div>
                    {% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}

                    <div class="form-group">
                        <div class="col-sm-3">

                        </div>
                        <div class="col-sm-9">
                            {% raw %}{{{% endraw %} submit_button("Save", "class": "pull-right btn btn-success") {% raw %}}}{% endraw %}
                        </div>
                    </div>
                {% raw %}{{{% endraw %} endForm() {% raw %}}}{% endraw %}
            </div>
        </div>
    </div>
</div>
{% raw %}{%{% endraw %} endblock {% raw %}%}{% endraw %}

Second example where a loop index is used as a counter:


{% raw %}{%{% endraw %} block content {% raw %}%}{% endraw %}
    {% raw %}{{{% endraw %} form("auth/login") {% raw %}}}{% endraw %}
        {% raw %}{{{% endraw %} form.render('csrf', ['value': security.getToken()]) {% raw %}}}{% endraw %}

        {% raw %}{%{% endraw %} for element in form {% raw %}%}{% endraw %}
            <div class="form-group has-feedback">
                {% raw %}{%{% endraw %} if loop.index == 1 {% raw %}%}{% endraw %}
                    {% raw %}{{{% endraw %} element {% raw %}}}{% endraw %}
                    <span class="glyphicon glyphicon-envelope form-control-feedback"></span>
                {% raw %}{%{% endraw %} elseif loop.index == 2 {% raw %}%}{% endraw %}
                    {% raw %}{{{% endraw %} element {% raw %}}}{% endraw %}
                    <span style="pointer-events: visible !important;" class="glyphicon glyphicon-eye-open form-control-feedback"></span>
                {% raw %}{%{% endraw %} endif {% raw %}%}{% endraw %}
            </div>
        {% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
        <div class="row">
            <div class="col-xs-8">
              {#<div class="checkbox icheck">
                <label>
                  <input type="checkbox" name="remember" > Remember
                </label>
              </div>#}
            </div><!-- /.col -->
            <div class="col-xs-4">
              {% raw %}{{{% endraw %} submit_button("Login", "class": "btn btn-primary btn-block btn-flat") {% raw %}}}{% endraw %}
            </div><!-- /.col -->
        </div>
    {% raw %}{{{% endraw %} endForm() {% raw %}}}{% endraw %}

    <a href="#">I forgot my password</a><br>
{% raw %}{%{% endraw %} endblock {% raw %}%}{% endraw %}
