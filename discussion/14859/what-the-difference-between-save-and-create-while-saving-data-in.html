Modify my previous answer:

foreach($this->request->getPost('data') as $item) {
    $model = new MyModel();
    $model->assign($item);
    if(!$model->create()) {
        // error
        foreach($model->getMessages() as $msg) {
            if($msg->getType() != 'Uniqueness') {
                // handle any error other than unique
                $this->flashSession->error($msg->getMessage());
            }
        }
    } else {
        // success
    }
}
