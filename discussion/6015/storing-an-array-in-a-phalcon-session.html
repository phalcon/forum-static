You can use the session like:

$session->person = ['name' => 'john', 'age' => 25];
//or
$session->set('person', ['name' => 'john', 'age' => 25]);

$session->person['age']++;

echo $session->person['age'];
//26

$session->person['age'] = 27;

print_r($session->person);
//Array ( [name] => john [age] => 27 )
