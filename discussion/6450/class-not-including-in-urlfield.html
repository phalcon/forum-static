The tag for stylesheet should be styleSheetInclude:


    echo Phalcon\Tag::stylesheetLink("https://fonts.googleapis.com/css?family=Rosario", false);
    echo Phalcon\Tag::stylesheetLink("css/style.css");

    //volt
    {% raw %}{{{% endraw %} stylesheet_link("https://fonts.googleapis.com/css?family=Rosario", false) {% raw %}}}{% endraw %}
    {% raw %}{{{% endraw %} stylesheet_link("css/style.css") {% raw %}}}{% endraw %}
