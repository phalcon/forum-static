By default router have only routes for /controller/action and /controller/action/params. You just need to add:

$router->add('/:module/:controller/:action',
        array(
            'module' => 1,
            'controller' => 2,
            'action' => 3,
        )
    );
    $router->add('/:module/:controller/:action/:params',
        array(
            'module' => 1,
            'controller' => 2,
            'action' => 3,
            'params' => 4
        )
    );
