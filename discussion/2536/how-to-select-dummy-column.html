You can do that with PHQL:

$phql = "SELECT m.*, 0 as status FROM Mymodel AS m";
$rows = $manager->executeQuery($phql);
foreach ($rows as $row) {
    echo $row->someField . "\n";
    echo $row->status . "\n";  //0
}
