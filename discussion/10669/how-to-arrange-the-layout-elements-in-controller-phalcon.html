The question is quite vague, but I hope this will help:

// controllers/Example.php
class ExampleController {
    public function someAction() {
        $this->view->tpl_header = 'header1';
        $this->view->tpl_left = 'left1';
        $this->view->tpl_center = 'center1';
        $this->view->tpl_right = 'right1';
        $this->view->data_center = [
            'some' => 'value',
        ];
    }
}
{# example/some.volt #}
<div class="header">
    {% raw %}{{{% endraw %} partial('partials/'~tpl_header) {% raw %}}}{% endraw %}
</div>
<div style="float:left; width: 160px;">
    {% raw %}{{{% endraw %} partial('partials/'~tpl_left) {% raw %}}}{% endraw %}
</div>
<div style="float:left; width: 400px;">
    {% raw %}{{{% endraw %} partial('partials/'~tpl_center, data_center) {% raw %}}}{% endraw %}
</div>
<div style="float:left; width: 160px;">
    {% raw %}{{{% endraw %} partial('partials/'~tpl_right) {% raw %}}}{% endraw %}
</div>
