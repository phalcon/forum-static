That should work, or you can use Phalcon\Mvc\Model\Behavior\Timestampable : 

public function initialize()
{
    $this->addBehavior(new Timestampable([
        'beforeSave'  => [
            'field'     => 'start',
            'format'    => 'Y-m-d H:i:s'
        ]
    ]));
}
