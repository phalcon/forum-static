Try register namespace or directory:

$loader->registerNamespaces(
    array(
        'Ecommerce\ExtendedModels' => '<path_to_app_ExtendModels_dir>',
    )
);
// or
$loader->registerDirs(array(
    '<path_to_app_ExtendModels_dir>'
))->register();
