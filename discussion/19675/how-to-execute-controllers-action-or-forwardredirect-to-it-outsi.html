dispatcher:beforeException I think is the right way. Now a not fancy way 

try {
    /////// Preparing something for app
   $response = $application->handle($_SERVER['REQUEST_URI']);
   $response->send();
} catch(Exception $e) {
   $application->handle('/a/not/fancy/url');
}
