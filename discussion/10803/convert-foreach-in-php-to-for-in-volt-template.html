{% raw %}{%{% endraw %} set i = 1 {% raw %}%}{% endraw %}
{% raw %}{%{% endraw %} for value in arr {% raw %}%}{% endraw %}
{% raw %}{{{% endraw %} i {% raw %}}}{% endraw %}
{% raw %}{%{% endraw %} set i += 1 {% raw %}%}{% endraw %}
{% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
