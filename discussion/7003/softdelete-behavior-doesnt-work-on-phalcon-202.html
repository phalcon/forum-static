<?php
// model
class Directories extends Model
{
    // ...
    public function initialize()
    {
        $this->addBehavior(new SoftDelete(
            [
                'field' => 'active',
                'value' => 0,
            ]
        ));
    }
    // ...
}

// and use
Directories::findFirst(1)->delete();
