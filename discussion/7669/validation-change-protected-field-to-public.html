A little idea u can use- new function in model: (example is from here https://docs.phalcon.io/en/latest/reference/validation.html for more details)

    public function getMessages($filter = NULL)
    {
        $messages = [];

        foreach (parent::getMessages($filter) as $message) {
            switch ($message->getType()) {
                case 'PresenceOf':
                    $messages[] = $message->getField().": Field is required");
                    break;
                default:
                    $messages[] = $message->getMessage();
            }
        }

        return $messages;
    }
