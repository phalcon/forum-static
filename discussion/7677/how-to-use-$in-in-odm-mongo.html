Got the right syntax. Here's the code for the benefit of others.

$products = Products::find(array(
    array(
        'tags' => array('$in' => array('beauty')),
        '$or' => array(
                    array('endDateTime' => ''),
                    array('endDateTime' => array('$gt' => new Mongodate(strtotime("now"))))
        )               
    ),
    "sort"  => array("_id" => -1)
));
