Try to use this im using it on my micro rest api 

        $loader = new Loader();
        $loader->registerNamespaces(array(
          'Controllers' => __DIR__.'/controllers/',
          'Models' => __DIR__.'/models/',
          'MyPackage\Robots' => __DIR__.'/../packages/appstack/robots/src'
        ))->register();
