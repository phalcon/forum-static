Like this:


      $form->add(new Select( "client" , Clients::find(
              [
                  'columns' => ['CONCAT(firstname, " ", lastname) as fullname', 'id']       # This is for MySQL database
              ]
        ) ,
          [
          'using'           =>  ['id','fullname'], 
          'data-placeholder' => "Members",
          'class'            => "chosen-select"
          ]
        )
      );
