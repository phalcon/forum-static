Validation slug for all models:

public function validation()
{  
    $validator = new Validation();

    $validator->add('slug', new UniquenessValidator(
        [
            "model"   => new Article(),
            "message" => "Slug is already exists",
        ]
    ));
    return $this->validate($validator);
}
