I add new function into /App/Services.php it work for me.

<?php
    protected function initRouter(){
        $router = new Router();
        $router->add("/:controller/:action", array(
        'controller' => 1,
        'action' => 2)); 
        $router->add("/tao", array(
        "controller" => "images",
        "action"     => "custom_route"
         ));
         return $router;
}
