private function reOrder($menuId)
    {  
        $menu = Menus::findFirst($menuId); 
        $position = 1;

        foreach (Menus::find(array( 'parentId = :parentId:', 'bind' => array( 'parentId' => $menu->parentId), 'order' => 'position ASC')) as $m)

            $nm->update();
