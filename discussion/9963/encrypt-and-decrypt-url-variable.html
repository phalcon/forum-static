Sure:

// Template file
<a href="{% raw %}{{{% endraw %} url(['for':'spm_contacto/edit','params':crypt.encrypt(id)]) {% raw %}}}{% endraw %}">Edit</a>
// Controller file
pubilc function editAction($encryptedId) {
    $id = $this->crypt->decrypt($encryptedId);
}
