This must work:

            "service" => "myViewCache",

This is what works for me:


    public function indexAction()
    {
        $this->view->cache(array(
            "service" => "myViewCache",
            "key"     => 'test'
        ));

        $exists = $this->view->getCache()->exists('test');
        if (!$exists) {
            $this->view->setTemplateAfter('main');
        } else {
            $this->view->cache(true);
        }
    }
