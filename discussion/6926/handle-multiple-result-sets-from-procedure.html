just foound a way to my question

        $result = $this->db->query("CALL procedure_name($param)");
        $result->setFetchMode(\PDO::FETCH_ASSOC);
        $resultArray=[];
        $resultArray['result1']=$result->fetchAll();
        $result->nextRowset();
        $resultArray['result2']=$result->fetchAll();
        $result->nextRowset();
