I think this argument is true by default, this is to indicate if the url is local (relative) or external (absolute)

$URL = new Phalcon\Mvc\Url;
// If you are on http://example.com/XX/
echo $URL->get('google.com', array(), false); //   CC/AA .. local relative to the current URL, http://example.com/XX/google.com
echo $URL->get('google.com', array(), true); //   /CC/AA .. local absolute to the current Domain, http://example.com/google.com
// Adding HTTP://
echo $URL->get('http://google.com', array(), false); //  external absolute URL, http://google.com
echo $URL->get('http://google.com', array(), true); //   local relative to the current URL http://example.com/XX/CC/AA/http://google.com
// So from the results you can see which links is true or not
