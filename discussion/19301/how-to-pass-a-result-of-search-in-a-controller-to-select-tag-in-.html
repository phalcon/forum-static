in controller

$this-view->setVar('providers', $providers);

https://docs.phalcon.io/3.4/en/views#value-transfer

echo $this->tag->Select(
  [
    'prodID',
    $providers, // variable from controller , to populate select tag
    'using' => [
      'id',
      'name',
    ],
    'id' => 'myId',
    'name' => 'myName'
  ]
);
