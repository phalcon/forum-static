I've tested and it's working:

class Robots extends Phalcon\Mvc\Model
{

        public function initialize()
        {
                $this->skipAttributesOnCreate(array('name'));
        }
}
$robot = new Robots();
$robot->name = 'hello';
$robot->type = 'some';
$robot->year = 1011;
var_dump($robot->save());
