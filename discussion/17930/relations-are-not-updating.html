class Categories2Articles extends Model
{
    /**
     * ID
     * @var integer
     */
    public $id;

    /**
     * Name
     * @var string
     */
    public $category_id;

    /**
     * Slug
     * @var string
     */
    public $article_id;

    /**
     * Sets table name
     *
     * @return string
     */
    public function getSource(): string
    {
        return "categories2articles";
    }

    /**
     * Define relationships to Users and Permissions
     */
    public function initialize()
    {
        $this->belongsTo(
            'category_id',
            __NAMESPACE__ . '\Categories',
            'id', [
                'alias' => 'categories',
                'reusable' => true
            ]
        );

        $this->belongsTo(
            "article_id",
            __NAMESPACE__ . '\Articles',
            'id', [
                'alias' => 'articles',
                'reusable' => true
            ]
        );
    }
}
