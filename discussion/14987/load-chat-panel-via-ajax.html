Got it!

Controller

public function chatPanelAction(){
    $id = $this->request->get("chatUserId");
    $user = $this->view->chat = Users::findFirst($id);
    $this->view->setVar('user',$user);
    $this->view->setRenderLevel(
        View::LEVEL_ACTION_VIEW
    );
}
