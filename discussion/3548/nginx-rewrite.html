Here's how I serve phalcon in a subfolder with nginx:

location @phalcon {
    rewrite ^/jiazu/public(.+)$ /jiazu/public/index.php?_url=$1 last;
}

location /jiazu/ {
    index index.php;
    if ($uri !~ ^/jiazu/public) {
        rewrite ^/jiazu(.*)$ /jiazu/public$1;
    }
    try_files $uri $uri/ @phalcon;
}
