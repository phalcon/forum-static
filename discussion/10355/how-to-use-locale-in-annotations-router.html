I want to get for frontend module https://localhost/en/controllerName/action. I use annotations router: php

$routerAnnotation = new Router\Annotations(false);  
$routerAnnotation->setDefaultModule('frontend');
$routerAnnotation->setDefaultNamespace('Modules\Frontend\Controllers');
$routerAnnotation->notFound(array("controller" => "index", "action" => "route404"));
$routerAnnotation->setUriSource(Router::URI_SOURCE_SERVER_REQUEST_URI);
$routerAnnotation->removeExtraSlashes(true);
$routerAnnotation->addModuleResource('frontend', 'Modules\Frontend\Controllers\Index', '/{locale:[a-z]{2,3{% raw %}}}{% endraw %}');
$routerAnnotation->addModuleResource('admin', 'Modules\Admin\Controllers\Index', '/admin');
$routerAnnotation->addModuleResource('admin', 'Modules\Admin\Controllers\User', '/admin/user');
$routerAnnotation->addModuleResource('frontend', 'Modules\Frontend\Controllers\Auth', "/{locale:[a-z]{2,3{% raw %}}}{% endraw %}/auth");

//IndexController
/**
 * @RoutePrefix("/")
*/
class IndexController extends Controller
{
        /**
        * @Get("/")
        */
public function indexAction()
{

}

For url https://localhost/en/ no routes was matched.
