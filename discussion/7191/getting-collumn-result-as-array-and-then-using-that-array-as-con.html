Got it to work, yay!


    public function newsfeed(){
        $IDs = array();
        $statuses = array();
        foreach($this->following as $followed){
             array_push($IDs, $followed->followed_id);
        }
        $IDs[] = $this->id;
        $statuses = Statuses::find([
            "conditions" =>
                "user_id = " . implode(" OR ",$IDs),
            "order" => "id DESC",
                ]);
        return $statuses;
    }
