Try the following:

cd cphalcon
cd build/32bits
export CC="gcc"
export CFLAGS="-O2 -fno-delete-null-pointer-checks -finline-functions -fomit-frame-pointer"
phpize
./configure --enable-phalcon
make && sudo make install
