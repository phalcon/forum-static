answer
Jan '15 
$di->set("sessionlogic", "SessionLogic");

$loader->registerDir(
    ["your/path" => "your/path/sessionlogic"]
);

or

$loader->registerNameSpace(
    ["your/namespace" => "your/path/sessionlogic"]
);
