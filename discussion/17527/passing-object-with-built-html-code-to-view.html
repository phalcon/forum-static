Im not 100% sure what you want to achieve, but maybe this example could help?

// Controller code
$this->view->users = User::find();

// Volt
<table>
    <thead>
        <tr>
            <th>ID</th>
            <th>Name</th>
            <th>Age</th>
        </tr>
    </thead>    
    <tbody>
        {% raw %}{%{% endraw %} for user in users {% raw %}%}{% endraw %}
        <tr>
            <td>{% raw %}{{{% endraw %} user.id {% raw %}}}{% endraw %}</td>
            <td>{% raw %}{{{% endraw %} user.name {% raw %}}}{% endraw %}</td>
            <td>{% raw %}{{{% endraw %} user.age {% raw %}}}{% endraw %}</td>
        </tr>
        {% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
    </tbody> 
</table>
