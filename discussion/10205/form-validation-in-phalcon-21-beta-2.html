validators in 2.1 are :


use Phalcon\Validation;

public function validation() {
$validator = new Validation();

$validator->add(
            "number",
            new Numericality([
                "model" : $this,
                "message" : "Enter a number"
            ])
        );
        return $this->validate();
    }
