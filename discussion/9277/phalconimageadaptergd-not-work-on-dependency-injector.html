Please try

$di->set('image', [
    'className' => 'Phalcon\Image\Adapter\GD',
    'arguments' => [
        [
            'type'  => 'parameter',
            'value' => 'path/image.jpg' // default parameter may be a null I think
        ]
    ]
]);

Calling DI

$di->getService('image')->setParameter(0, array(
    'type'  => 'parameter',
    'value' => 'path/image.jpg'
))->resize(200,200)->save();
