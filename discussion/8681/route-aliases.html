foreach ($application->getModules() as $key => $module) {
                    $router->add('/'.$key.'/:params', array(

                        'controller' => 'index',
                        'action' => 'index',
                        'params' => 1
                    ))->setName($key);

                    $router->add('/'.$key.'/:controller/:params', array(
                        'module' => $key,
                        'controller' => 1,

                    $router->add('/'.$key.'/:controller/:action/:params', array(
