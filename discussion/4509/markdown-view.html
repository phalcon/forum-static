    $view = new \Phalcon\Mvc\View();

    $view->registerEngines([
        '.md' => function($view, $di) {
            $md = new \Phalcon\Mvc\View\Engine\Markdown($view, $di);
            $md->setOptions([
                'compiledPath' => '../app/compiled-templates/',
                'compiledSeparator' => '_',
