Ok.

I locate JS files in myapp/public/vendor/*

.htaccess 

<IfModule mod_rewrite.c>
    RewriteEngine on
    RewriteRule  ^$ public/    [L]
    RewriteRule  (.*) public/$1 [L]
</IfModule>

public/.htaccess 

AddDefaultCharset UTF-8

<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteCond %{REQUEST_FILENAME} !-d
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteRule ^(.*)$ index.php?_url=/$1 [QSA,L]
</IfModule>

DI config in services.php 

$di->set(
    'url',
    function () use ($config) {
        $url = new UrlResolver();
        $url->setBaseUri($config->application->path); // /myapp/
        $url->setStaticBaseUri($config->application->staticPath); // https://localhost/myapp/

        return $url;
    }
);

Use in volt 

{% raw %}{%{% endraw %} block js {% raw %}%}{% endraw %}
    {% raw %}{%{% endraw %} do assets.collection('vendor-js').addJs('vendor/jquery/dist/jquery.min.js') {% raw %}%}{% endraw %}
    {% raw %}{%{% endraw %} do assets.collection('vendor-js').addJs('vendor/bootstrap/dist/js/bootstrap.min.js') {% raw %}%}{% endraw %}
{% raw %}{%{% endraw %} endblock {% raw %}%}{% endraw %}
{% raw %}{{{% endraw %} assets.outputJs('vendor-js') {% raw %}}}{% endraw %}

In my mind this elements are most important.
