you can create a custom "findForSelect" function in your model, where you do the aggregate

    public static function findForSelect()
    {
      $users = parent::find(array(
                  "columns" => (
                  array(
                      "id",
                      "surname",
                      "name")
                  )
              ));
              $length = count($users);
              $unames = array();
              for($i=0;$i<$length;$i++) {
                  $unames[$users[$i]['id']] = $users[$i]['prenom'] . " " . $users[$i]['nom'];
              }

              return $unames;
          }
