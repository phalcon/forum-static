class MyUrl extends Phalcon\Mvc\Url
{
    public function get($uri = null, $args = null, $local = null, $baseUri = null) {
        $lang = // get language from somwhere
        $name = "$uri-$lang";
        $params = array_merge (['for' => $name], $args);
        return parent::get($params)
    }
}
