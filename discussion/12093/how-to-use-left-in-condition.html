this seems to be working for me : 


        $di = \Phalcon\Di::getDefault();

        $db = $di->get('db');

        $sql = $db->query("SELECT * FROM `delete_it` WHERE LEFT(dat, 4) = 2015");

        $res = $sql->fetchAll();

    CREATE TABLE `delete_it` (
