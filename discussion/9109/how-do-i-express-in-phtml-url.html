volt

{% raw %}{{{% endraw %} link_to("spmcontacto/new", "<img src='"~url("img/tolbar/Package-accept-icon.png")~"'> Anadir un nuevo Empleado) {% raw %}}}{% endraw %}

phtml

<?php echo $this->tag->linkTo("spmcontacto/new", "<img src='" . $this->url->get("img/tolbar/Package-accept-icon.png") . "'> Anadir un nuevo Empleado") ?>
