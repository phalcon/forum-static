Models:

User

$this->hasMany('_id', 'UserRole', 'user_id', array('alias' => 'UserRole'));

Role

$this->hasMany('_id', 'UserRole', 'role_id', array('alias' => 'UserRole'));

  1. $this->belongsTo('role_id', 'Role', '_id', array('alias' => 'Role'));
