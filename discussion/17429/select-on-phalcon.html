// Here Categories is some model
$categories = Categories::find([
    'conditions' => 'number_posts BETWEEN ?1 AND ?2',
    'order'      => 'name DESC',
    'limit'      => 10,
    'bind'       => [
        1 => 8,
        2 => 12,
    ],
]);

$select = Phalcon\Tag::select(
    [
        'category',
        $categories,
        'name'      => 'categories',
        'emptyText' => 'Choose category ...',
        'useEmpty'  => true,
        'using'     => [
            'id',
            'name'
        ],
    ]
);

echo $select;
