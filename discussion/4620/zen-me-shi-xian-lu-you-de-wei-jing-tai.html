$di->set('router', function() {
        $router = new \Phalcon\Mvc\Router(false);
        //不知道你是这个意思么？
        $router->add(
            "/test/signup.html",
            array(
                "controller" => "signup",
                "action"     => "test",
            )
        );
        $router->add(
            "/test/register.html",
            array(
                "controller" => "signup",
                "action"     => "register",
            )
        );
        return $router;
    },true);
    // Handle the request
    $application = new Application($di);
