            SELECT a, COUNT(a) FROM
            (SELECT user_id, a FROM table GROUP BY user_id) as users

        return new \Phalcon\Mvc\Model\Resultset\Simple(
            null,
            $table,
            $table->getReadConnection()->query($sql)
