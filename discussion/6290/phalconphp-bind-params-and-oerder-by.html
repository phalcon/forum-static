you must define mandatory key order and limit, 

$conditions = "category = :id: AND status = :status: ";
$parameters = array( "id" => $cat_id, "status" => 1 );
$posts = Posts::find(array( 
    $conditions, 
    "bind" => $parameters, 
    "order" => "order DESC", 
    "limit" => "30" 
));
