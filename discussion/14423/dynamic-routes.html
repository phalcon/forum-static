Good day :)

My decision

  // for /v1/logins/$id:[0-9]
  function checkIgnored($app) {
      if(preg_match('/[(\W|^)logins(\W|$)]\d+/', $app->router->getRewriteUri(), $matches)) {
          return str_replace($matches[0], '', $app->router->getRewriteUri());
      } else {
          return $app->router->getRewriteUri();
      }
  }
