            <a href="{% raw %}{{{% endraw %} url('show-app/'~app.getSlugApp()) {% raw %}}}{% endraw %}">
                {% raw %}{{{% endraw %} app.getNameApp() {% raw %}}}{% endraw %}
                <p>{% raw %}{{{% endraw %} position {% raw %}}}{% endraw %}</p>

                    {% raw %}{{{% endraw %} image('files/imgs/icon/'~app.getUrlIconApp() ) {% raw %}}}{% endraw %}
                {% raw %}{%{% endraw %} else {% raw %}%}{% endraw %}
                    {% raw %}{{{% endraw %} image('files/imgs/default_icon.png') {% raw %}}}{% endraw %}
                {% raw %}{%{% endraw %} endif {% raw %}%}{% endraw %}

                {% raw %}{%{% endraw %} for ai in app.getAppInfo() {% raw %}%}{% endraw %}

                        <div class="subheader">{% raw %}{{{% endraw %} ai.getAcessAppInfo() {% raw %}}}{% endraw %} acessos</div>
                    {% raw %}{%{% endraw %} endif {% raw %}%}{% endraw %}
                {% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
            </a>
        </li>
    {% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
</ul>
