Try returning the connection directly:

class Menu extends \Phalcon\Mvc\Model
{
   public function getWriteConnection()
   {
       return \Phalcon\DI::getDefault()->get('db_cms');
   }
   public function getReadConnection()
   {
       return \Phalcon\DI::getDefault()->get('db_cms');
   }
