If you use a Micro MVC yes: http://docs.phalcon.io/en/latest/reference/micro.html

<?php

$app = new Phalcon\Mvc\Micro();

$app->get('/say/welcome/{name}', function ($name) {
    echo "<h1>Welcome $name!</h1>";
});

$app->handle();
