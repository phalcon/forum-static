Now it is working i have changed class in div to form-group has-feedback.

                        <!-- form start -->
                        {% raw %}{{{% endraw %} form("expense/createExpense") {% raw %}}}{% endraw %}
                        <div class="box-body">
                            {% raw %}{%{% endraw %} for element in form {% raw %}%}{% endraw %}
                                {% raw %}{%{% endraw %} if is_a(element, 'Phalcon\Forms\Element\Hidden') {% raw %}%}{% endraw %}
                                    {% raw %}{{{% endraw %} element {% raw %}}}{% endraw %}
                                {% raw %}{%{% endraw %} elseif element.getName() == "txn_date" {% raw %}%}{% endraw %}
                                    <div class="form-group has-feedback" >
                                        {% raw %}{{{% endraw %} element.label() {% raw %}}}{% endraw %}
                                        {% raw %}{{{% endraw %} element.render(['class': 'form-control']) {% raw %}}}{% endraw %}
                                        <span class="glyphicon glyphicon-calendar form-control-feedback"></span>
                                    </div>
                                {% raw %}{%{% endraw %} else {% raw %}%}{% endraw %}
                                    <div class="form-group" >
                                        {% raw %}{{{% endraw %} element.label() {% raw %}}}{% endraw %}
                                        {% raw %}{{{% endraw %} element.render(['class': 'form-control']) {% raw %}}}{% endraw %}
                                    </div>
                                {% raw %}{%{% endraw %} endif {% raw %}%}{% endraw %}
                            {% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
                            <div align="center">
                                {% raw %}{{{% endraw %} submit_button("Create", "class": "btn btn-primary") {% raw %}}}{% endraw %}
                            </div>

                        </div>
                        {% raw %}{{{% endraw %} end_form() {% raw %}}}{% endraw %}
