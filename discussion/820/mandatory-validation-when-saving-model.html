You can try the skipAttributes() in your model's initialize().

class User extends \Phalcon\Mvc\Model
{
    public function initialize()
    {
        $this->skipAttributes(
            [
                'field1',
                'field2',
                'field3',
                'field4',
            ]
        );
    }
}
