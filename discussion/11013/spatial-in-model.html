$poi = Poi::find(array(
        'conditions' => "INTERSECTS(geopoint, GeomFromText(?1))",
        'bind' => [
                1 => "POLYGON(($polyData))",
        ],
        'columns' => [
            'lon' => 'X(geopoint)',
            'lat' => 'Y(geopoint)',
            'title',
            'id',
            'address',
        ],
        'limit' => 20,
        'order' => 'rand()'
));
