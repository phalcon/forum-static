$this->cookies->set('RMT', $token, $expire, "/", false, $this->config->cookie->defaultDomain);

    if ($this->cookies->has('RMU')) {
        $this->cookies->get('RMU')->delete();

    if ($this->cookies->has('RMT')) {
        $this->cookies->get('RMT')->delete();

What's the problem?
