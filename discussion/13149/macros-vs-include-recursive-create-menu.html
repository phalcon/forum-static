I create a tree menu with:

<!--includes/menu-links.html-->
{% raw %}{%{% endraw %} for link in links {% raw %}%}{% endraw %}
    <li>
        <a href="{% raw %}{{{% endraw %} link.href {% raw %}}}{% endraw %}">{% raw %}{{{% endraw %} link.name {% raw %}}}{% endraw %}</a>
        {% raw %}{%{% endraw %} if link.sublinks {% raw %}%}{% endraw %}
            <ul>
                {% raw %}{%{% endraw %} include "includes/menu-links.html" with ['links': link.sublinks] {% raw %}%}{% endraw %}
            </ul>
        {% raw %}{%{% endraw %} endif {% raw %}%}{% endraw %}
    </li>
{% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}

call:

<ul class="main-menu">
    {% raw %}{%{% endraw %} include "includes/menu-links.html" with ['links':links] {% raw %}%}{% endraw %}
</ul>

or

<!--macros/menu-macros.html-->
{% raw %}{%{% endraw %} macro menu_links(links) {% raw %}%}{% endraw %}
    {% raw %}{%{% endraw %} for link in links {% raw %}%}{% endraw %}
        <li>
            <a href="{% raw %}{{{% endraw %} link.href {% raw %}}}{% endraw %}">{% raw %}{{{% endraw %} link.name {% raw %}}}{% endraw %}</a>
            {% raw %}{%{% endraw %} if link.sublinks {% raw %}%}{% endraw %}
                <ul>
                    {% raw %}{{{% endraw %} menu_links(link.sublinks) {% raw %}}}{% endraw %}
                </ul>
            {% raw %}{%{% endraw %} endif {% raw %}%}{% endraw %}
        </li>
    {% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
{% raw %}{%{% endraw %} endmacro {% raw %}%}{% endraw %}

call:

<ul class="main-menu">
    {% raw %}{{{% endraw %} macros.menu_links(links) {% raw %}}}{% endraw %}
</ul>

What approach would have a better performance?
