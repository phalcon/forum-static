You can use an alias this way:

 $builder = $this->modelsManager->createBuilder()
                ->from(array('la' => '\Core\Model\LanguageTranslation'))
                ->where("la.original LIKE '%{$search}%'")
                ->orWhere("la.translated LIKE '%{$search}%'");
