    {
        protected $db;

            $this->db = $this->getDi()->getShared('db');
        }

Maybe this is what you want:

<?php
use Phalcon\DI;
use Phalcon\Mvc\Model;

class BaseModel extends Model
{
    protected $db;

    public function initialize()
    {
        $this->db = DI::getDefault()->get('db');
    }
}
