Try:

$di->set('cache', function(){
    $front = new Phalcon\Cache\Frontend\Data([]);
    $cache = new Phalcon\Cache\Backend\Apc($front, []);
    return $cache;
}, true);

And:

$di = \Phalcon\DI::getDefault();

$cache = $di['cache'];
