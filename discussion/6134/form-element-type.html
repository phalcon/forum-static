My solution is:

    public function beforeValidation($data, $entity, $messages)
    {
        $elements = $this->getElements();
        foreach ($elements as $element){
            $is_checkbox = is_a($element, 'Phalcon\Forms\Element\Check');
            if ($is_checkbox){
                // do stuff
            }
        }
    }
