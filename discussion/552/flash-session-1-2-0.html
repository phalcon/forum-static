Return the response object or disable the view rendering:

public function A()
{
    $this->flashSession->success('You did it !');
    return $this->response->redirect('b');
}
public function A()
{
    $this->flashSession->success('You did it !');
    $this->view->disable();
    $this->response->redirect('b');
}
