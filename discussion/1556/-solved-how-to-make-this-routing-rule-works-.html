move the logic to action method.

        $router->add('/{prefix:[A-Z]rows}/{partner:[a-zA-Z0-9]{3{% raw %}}}{% endraw %}', ['controller' => 'Rows',   'action' => 'handler'])
            ->convert('partner', function($partner) {
                return $prefix . '_' . $partner;
            });

or

        $router->add('/Rows/{prefix:[A-Z]}/{partner:[a-zA-Z0-9]{3{% raw %}}}{% endraw %}', ['controller' => 'Rows',   'action' => 'handler'])
            ->convert('partner', function($partner) {
                return $prefix . '_' . $partner;
            });
