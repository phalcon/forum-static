    'action' => 2,
))->convert('action', function($action) {
    return Phalcon\Text::camelize($action);

// Route for controller action params (support dashes in actions)

    'namespace' => 'App\Controllers\\',
    'controller' => 1,

    'params' => 3

    return Phalcon\Text::camelize($action);
});
