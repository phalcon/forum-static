Using an AND?

if (!is_array($parameters)) {
 $parameters = array();
 }

$conditions = array();
 if($sessionid){
$conditions[] = 'sessionid = '.$sessionid;
 }
  if($listingid){
$conditions[] = 'listingid = '.$listingid;
 }
  if($sessionid){
$conditions[] = 'userid = '.$userid;
 }

 if (count($conditions)) {
     $parameters['conditions'] = join(' AND ', $conditions);
 }
