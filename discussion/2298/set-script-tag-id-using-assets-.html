I found your question when searching how to do this, so I had to find out myself :

$this->assets->addJs( "path/to/file.js", true, false,  array( "id" => "unique-js-id") );
{% raw %}{%{% endraw %} for resource in assets.collection('footer') {% raw %}%}{% endraw %}
   {% raw %}{{{% endraw %} javascript_include( [ 'src' : resource.getPath(), 'type' : 'text/javascript', 'id' : resource.getAttributes()['id']  ]  )  {% raw %}}}{% endraw %}
{% raw %}{%{% endraw %} endfor {% raw %}%}{% endraw %}
