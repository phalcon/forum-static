Try adding alias to relationship

In items


    public function initialize()
    {        
        $this->hasMany('id','\Apps\Main\Models\Comments','item_id', array('alias' => 'Comments'));
    }

In comments


    public function initialize()
    {
        $this->belongsTo('item_id','\Apps\Main\Models\Items','id', array('alias' => 'Items'));            
    }
