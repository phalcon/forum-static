I need to:

$numbers = Number::find();

$numbers = Number::find();
$result = [];
foreach($numbers as $n){
    $arr = $n->toArray();
    $arrOrder = ($n->order) ? $n->order->toArray() : null;
    $arr['order'] = $arrOrder;
    $result[] = $arr;
}

echo( json_encode($result));

class Number implements JsonSerializable
{
    public function jsonSerialize()
    {
        $ret = $this->toArray();
        $ret['order'] = ($this->order) ? $this->order->toArray() : null;
        return $ret;
    }
}
