        $frontendRoutes = 'routes_int.php';
        break;
}

return new \Phalcon\Config(
    array(
        'database' => array(

        'application' => array(

            'frontendRoutes' => $frontendRoutes,

$config = include __DIR__ . "/config.php";
require(__DIR__ . '/../apps/frontend/config/'.$config->app->frontendRoutes);
