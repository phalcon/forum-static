$di = new FactoryDefault();
$di->set('flash', function () {
    return new Flash(array(
        'error' => 'css_class',
        'error2' => 'css_class',
        'warning' => 'css_class',
        'success' => 'css_class',
        'notice' => 'css_class'
    ));
});
