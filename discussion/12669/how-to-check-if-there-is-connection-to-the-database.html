        $config = array(
        "host" => "192.168.0.11",
        "dbname" => "blog",
        "port" => 3306,
        "username" => "sigma",
        "password" => "secret"

        return $connection = new \Phalcon\Db\Adapter\Pdo\Mysql($config);

        }
        catch(Exception $e) {
            //dosomething
            return false;

in controller you can use also try - catch

try { $this->db; } catch (\Exception $e) { $this->logger->critical("cannot connect to database"); return; }
