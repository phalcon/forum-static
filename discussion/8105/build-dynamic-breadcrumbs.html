<ul>
{#
   <li><a href="{% raw %}{{{% endraw %} url(dispatcher.getModuleName()~'/index/index') {% raw %}}}{% endraw %}">{% raw %}{{{% endraw %} dispatcher.getModuleName() {% raw %}}}{% endraw %}</a></li>
#}
   <li><a href="{% raw %}{{{% endraw %} url(dispatcher.getModuleName()~'/'~dispatcher.getControllerName()~'/index') {% raw %}}}{% endraw %}">{% raw %}{{{% endraw %} dispatcher.getControllerName() {% raw %}}}{% endraw %}</a></li>
   <li>{% raw %}{{{% endraw %} dispatcher.getActionName() {% raw %}}}{% endraw %}</li>
</ul>
