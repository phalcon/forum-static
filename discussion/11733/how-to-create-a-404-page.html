This will forward the dispatcher to a desired controller+action:

$di->setShared('router' ,function() use($config) {
    $router = new \Phalcon\Mvc\Router();
    $router->notFound(array(
        "controller" => "index",
        "action" => "route404"
    ));
    return $router;
});
